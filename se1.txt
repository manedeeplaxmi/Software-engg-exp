
public class Cube {
	int side;
	double volume;

	public void volume() // volume calculation
	{
    	volume =  side * side * side;
    
	}

	public static void main(String arg[]) {
    	Cube c1 = new Cube();
    	c1.side = 5;
    	c1.volume(); // calling the function volume
    	System.out.println(c1.volume);
	}
}
